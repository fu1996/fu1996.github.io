function compose(...funcs) {
  return function (...args) {
    const lastFn = funcs[funcs.length - 1];
    const fns = funcs.slice(0, funcs.length - 1);
    return fns.reduceRight((result, fn) => fn(result), lastFn(...args));
  };
}
